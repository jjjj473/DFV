# JavaScript for Forms

const form = document.querySelector('form'); : Reference a form element.
form.addEventListener('submit', handleSubmit); : Intercept a form submission.
const input = document.querySelector('input'); : Reference a specific input.
input.value : Get or set the current value.
input.addEventListener('input', handleInput); : React to user typing.

const select = document.querySelector('select'); : Reference a drop-down list.
select.value : Retrieve the selected option value.
select.addEventListener('change', handleChange); : Respond when the option changes.

document.forms[0] : Access the first form in the document.
new FormData(form); : Collect all fields for submission.
event.preventDefault(); : Prevent the browser's default form submission.
form.reset(); : Clear user input and reset the form.
